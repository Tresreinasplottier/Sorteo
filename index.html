async function cargarNumeros() { const grilla = document.getElementById('grilla'); try { // Agregamos un número aleatorio al final de la URL para evitar el "cache" (datos viejos) const response = await fetch(scriptURL + '?v=' + Math.random(), { method: 'GET' }); if (!response.ok) throw new Error('Error en respuesta de red'); const data = await response.json(); grilla.innerHTML = ''; let reservados = 0; let total = data.length - 1; if (total <= 0) { grilla.innerHTML = "No hay números cargados en el Excel."; return; } for (let i = 1; i < data.length; i++) { const nro = data[i][0]; const estado = data[i][2]; if (estado === 'Reservado') reservados++; const div = document.createElement('div'); div.className = 'nro' + (estado === 'Reservado' ? ' reservado' : ''); div.innerText = nro; if (estado !== 'Reservado') { div.onclick = function() { document.querySelectorAll('.nro').forEach(el => el.classList.remove('seleccionado')); div.classList.add('seleccionado'); numeroSeleccionado = nro; document.getElementById('nro-elegido').innerText = nro; document.getElementById('form-reserva').style.display = 'block'; document.getElementById('form-reserva').scrollIntoView({ behavior: 'smooth', block: 'center' }); }; } grilla.appendChild(div); } // Actualizar barra let porcentaje = Math.round((reservados / total) * 100); document.getElementById('barra-progreso').style.width = porcentaje + "%"; document.getElementById('porcentaje-texto').innerText = porcentaje + "% COMPLETADO"; } catch (e) { console.error(e); grilla.innerHTML = "Error de conexión con el Excel. Revisá los permisos del Script."; } }
